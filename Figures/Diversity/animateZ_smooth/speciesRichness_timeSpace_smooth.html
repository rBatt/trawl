## Species richness over time and space
library(maps)
library(data.table)
library(animation)
par(mar = c(1.75, 1.5, 0.5, 0.5), oma = rep(0.1, 4), mgp = c(0.85, 
    0.05, 0), tcl = -0.15, ps = 8, family = "Times", cex = 1)
ani.options(inverval = 0.5)
smooZ[, j = {
    plot(lon, lat, type = "n", xlab = "", ylab = "", ylim = lim.lat, 
        xlim = lim.lon)
    invisible(map(add = TRUE, fill = TRUE, col = "lightgray"))
    points(lon, lat, pch = 21, bg = z.col)
    segments(x0 = -165, x1 = -160, y0 = seq(30, 40, length.out = 256), 
        col = heat.cols)
    segments(x0 = -166, x1 = -165, y0 = seq(30, 40, length.out = 4), 
        col = "black")
    text(-167, y = seq(30, 40, length.out = 4), round(seq(lim.z[1], 
        lim.z[2], length.out = 4), 2), adj = 1, cex = 1, 
        col = "black")
    text(-162.5, 41.5, bquote(Species ~ Richness))
    ani.pause()
}, by = c("year")]
